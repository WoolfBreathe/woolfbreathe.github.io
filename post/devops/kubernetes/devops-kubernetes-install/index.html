二、基础环境配置（以下操作所有节点都得执行）

2.1、配置hosts解析

修改主机名字

[root@k8s-master01 ~]# hostnamectl set-hostname k8s-master01

配置Host文件

[root@k8s-master01 ~]# cat >> /etc/hosts << EFO
127.0.0.1 localhost
127.0.1.1 k8s-master01

192.168.1.100 k8s-api.virtual.local k8s-api

192.168.1.100 k8s-master01
192.168.1.101 k8s-master02
192.168.1.102 k8s-master03
192.168.1.246 k8s-master-lb # 如果不是高可用集群，该IP为Master01的IP
192.168.1.103 k8s-node01
192.168.1.104 k8s-node02
EFO

环境变量

每个服务器的NODE_NAME和NODE_NAME需要单独修改

cat > /etc/profile.d/k8s-env.sh <<EFO
# TLS Bootstrapping 使用的Token，可以使用命令 head -c 16 /dev/urandom | od -An -t x | tr -d ' ' 生成
BOOTSTRAP_TOKEN="8981b594122ebed7596f1d3b69c78223"

# 建议使用未用的网段来定义服务网段和Pod 网段
# 服务网段(Service CIDR)，部署前路由不可达，部署后集群内部使用IP:Port可达
SERVICE_CIDR="10.96.0.0/12"

# Pod 网段(Cluster CIDR)，部署前路由不可达，部署后路由可达(flanneld 保证)
CLUSTER_CIDR="172.16.0.0/12"

# kubernetes 服务IP(预先分配，一般为SERVICE_CIDR中的第一个IP)
CLUSTER_KUBERNETES_SVC_IP="10.96.0.1"

# 集群 DNS 服务IP(从SERVICE_CIDR 中预先分配)
CLUSTER_DNS_SVC_IP="10.96.0.10"

# 集群 DNS 域名
CLUSTER_DNS_DOMAIN="cluster.local."

# MASTER API Server 地址
MASTER_URL="k8s-api.virtual.local"

KUBE_APISERVER="https://k8s-api.virtual.local:6443"

# 服务端口范围(NodePort Range)
NODE_PORT_RANGE="30000-32766"

# flanneld 网络配置前缀
FLANNEL_ETCD_PREFIX="/kubernetes/network"



# 当前部署的机器名称(随便定义，只要能区分不同机器即可)
NODE_NAME="k8s-master01"

# 当前部署的机器IP
NODE_IP="192.168.1.100"

# Etcd 集群所有机器 IP
EtcdNodes="k8s-master01 k8s-master02 k8s-master03"

# Master 集群所有机器 IP
MasterNodes="k8s-master01 k8s-master02 k8s-master03"

# Work 集群所有机器 IP
WorkNodes="k8s-node01 k8s-node02"

# CA 集群所有机器 IP
CANodes="10.96.0.1,10.96.0.10,127.0.0.1,192.168.1.100,k8s-api.virtual.local,k8s-api,kubernetes,kubernetes.default,kubernetes.default.svc,kubernetes.default.svc.cluster,kubernetes.default.svc.cluster.local,k8s-master01,k8s-master02,k8s-master03,k8s-node01,k8s-node02"




# etcd集群服务地址列表
ETCD_ENDPOINTS="https://192.168.1.100:2379,https://192.168.1.101:2379,https://192.168.1.102:2379"

# Etcd 集群间通信的IP和端口(注意:服务器名字要和IP对应)
ETCD_NODES="k8s-master01=https://192.168.1.100:2380,k8s-master02=https://192.168.1.101:2380,k8s-master03=https://192.168.1.102:2380"
EFO

Ubuntu

因为Gemfield的K8s集群即将部署的是calico网络插件，而calico需要这个内核参数是0或者1，但是Ubuntu20.04上默认是2，这样就会导致calico插件报下面的错误（这是个fatal级别的error）：

int_dataplane.go 1035: Kernel's RPF check is set to 'loose'.  \
This would allow endpoints to spoof their IP address.  \
Calico requires net.ipv4.conf.all.rp_filter to be set to 0 or 1. \
If you require loose RPF and you are not concerned about spoofing, \
this check can be disabled by setting the IgnoreLooseRPF configuration parameter to 'true'.

修改这个参数的值：

#修改/etc/sysctl.d/10-network-security.conf
gemfield@ai05:~$ sudo vi /etc/sysctl.d/10-network-security.conf

#将下面两个参数的值从2修改为1
net.ipv4.conf.default.rp_filter=1
net.ipv4.conf.all.rp_filter=1

重新加载内核参数

sysctl --system

七、生成证书

Kubernetes 系统各个组件需要使用TLS证书对通信进行加密，这里我们使用CloudFlare的PKI 工具集cfssl 来生成Certificate Authority(CA) 证书和密钥文件， CA 是自签名的证书，用来签名后续创建的其他TLS 证书。

$ cat ca-config.json
{
    "signing": {
        "default": {
            "expiry": "87600h"
        },
        "profiles": {
            "kubernetes": {
                "expiry": "87600h",
                "usages": [
                    "signing",
                    "key encipherment",
                    "server auth",
                    "client auth"
                ]
            }
        }
    }
}

config.json：可以定义多个profiles，分别指定不同的过期时间、使用场景等参数；后续在签名证书时使用某个profile； signing: 表示该证书可用于签名其它证书；生成的ca.pem 证书中CA=TRUE； server auth: 表示client 可以用该CA 对server 提供的证书进行校验； client auth: 表示server 可以用该CA 对client 提供的证书进行验证。

            "OU": "System"
        }
    ]
}

CN: Common Name，kube-apiserver 从证书中提取该字段作为请求的用户名(User Name)；浏览器使用该字段验证网站是否合法； O: Organization，kube-apiserver 从证书中提取该字段作为请求用户所属的组(Group)； hosts 字段指定授权使用该证书的节点IP

# 修改calico-etcd.yaml的以下位置
sed -i 's#etcd_endpoints: "http://<ETCD_IP>:<ETCD_PORT>"#etcd_endpoints: "${ETCD_ENDPOINTS}"#g' calico-etcd.yaml

sed -i "s@# etcd-key: null@etcd-key: ${ETCD_KEY}@g; s@# etcd-cert: null@etcd-cert: ${ETCD_CERT}@g; s@# etcd-ca: null@etcd-ca: ${ETCD_CA}@g" calico/calico-etcd.yaml

sed -i 's#etcd_ca: ""#etcd_ca: "/calico-secrets/etcd-ca"#g; s#etcd_cert: ""#etcd_cert: "/calico-secrets/etcd-cert"#g; s#etcd_key: "" #etcd_key: "/calico-secrets/etcd-key" #g' calico-etcd.yaml

# 更改此处为自己的pod网段
sed -i 's@# - name: CALICO_IPV4POOL_CIDR@- name: CALICO_IPV4POOL_CIDR@g; s@#   value: "192.168.0.0/16"@  value: '"${CLUSTER_CIDR}"'@g' calico-etcd.yaml
